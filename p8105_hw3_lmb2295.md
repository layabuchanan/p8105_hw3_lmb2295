``` r
accel_df =
  janitor::clean_names(accel_df) %>%  
  mutate(
    weekend = case_when(
      day == "Monday" ~ "no",
      day == "Tuesday" ~ "no",
      day == "Wednesday" ~ "no",
      day == "Thursday" ~ "no",
      day == "Friday" ~ "no",
      day == "Saturday" ~ "yes",
      day == "Sunday" ~ "yes",
      TRUE ~ ""
    )) %>%
   mutate(weekend = recode(weekend, "yes" = TRUE, "no" = FALSE)) %>% 
  relocate(weekend, .after = day)
  
accel_class_df = 
  accel_df %>% 
  summarise_all(typeof) %>% 
  select(week, day_id, day, weekend, activity_4)
```
